# define our function to get the stadium coordinates
def retrieve_lat_long(country, nTeams):
    
    # get wikipedia page HTML
    
    lead = "List of football stadiums in "
    lead = (re.sub('football ', '', lead) if country == 'Spain' else lead)
    country_text = wikipedia.page(lead + country).html()
    
    # get text and find the relevant column names we want
    soup = BeautifulSoup(country_text)
    cols = []
    name = soup.find('table', attrs = {'class':'wikitable sortable'}).find_all('th')
    [cols.append(i.text.replace('\n','')) for i in name]
    
    # specifically find column names for each countries' respective page
    if country == 'Spain':
        s = cols.index('Stadium')
        t = cols.index('Team')
    elif country == 'Italy':
        s = cols.index('Stadium')
        t = cols.index('Home Team')
    elif country == 'England':
        s = cols.index('Stadium')
        t = cols.index('Club')
    elif country == 'Germany':
        s = cols.index('Stadium')
        t = cols.index('Home team')
    
    # find the stadium info within the tables
    
    stadiums = []
    teams = []

    table = soup.find('table', attrs = {'class':'wikitable sortable'})

    # try to get the link to the stadiums wikipedia page (if it exists)
    tr = table.find_all('tr')
    for row in tr[1:]:
        try:
            stadiums.append(row.find_all('td')[s].find('a').get('href'))
            teams.append(row.find_all('td')[t].text)
        except:
            pass

    # as the links are fragments, create the full URL for the team and stadium    
    links = ['https://en.wikipedia.org' + s for s in stadiums]
    teams = [t.replace('\n', '') for t in teams]
    output = pd.DataFrame(list(zip(teams, links)), columns = ['Team', 'Stadium'])
    
    # because some countries have many junior divisions, limit the number of stadiums to 
    # either the total number present, or the first N results (whichever is lower)
    nTeams = min(nTeams, output.shape[0])
        
    output = output.head(nTeams)
    
    # get stadium lat-long pairs
    
    lat = []
    lon = []
    
    # loop through the stadium wikipedia pages and look for the coordinate classes
    for i in range(nTeams):
        try:
            html = requests.get(output.Stadium[i])
            soup = BeautifulSoup(html.text, 'html.parser')
            lat.append(soup.find(class_ = 'latitude').text)
            lon.append(soup.find(class_ = 'longitude').text)
            
        except:
            lat.append(np.NaN)
            lon.append(np.NaN)
            print('Skipped: ', output.Team[i])
    
    output['Lat'] = lat
    output['Lon'] = lon

    output = output[output.Team != 'None']
    print('complete')
    
    # return our data
    return(output)

# convert the coordinates from degrees, minutes, seconds to fractions of degrees
# for use in plotly
def coord_convert(x):
    try:
        loc = re.split('[^\d\w\.]+', x)
        degree = round(float(loc[0]) + (float(loc[1]) / 60) + (float(loc[2]) / 3600),6)
        hem = (1 if loc[3] in ['N', 'E'] else -1)
        return(degree * hem)
    except:
        return np.NaN
